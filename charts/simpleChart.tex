\documentclass[border=15pt]{standalone}
\usepackage{starfont}
\usepackage{tikz}
\usetikzlibrary{backgrounds}

\begin{document}

% draw the chart circle
\begin{tikzpicture}[scale=4.3, rotate=180]
        
        % draw the circles
        \draw[thick] (0,0) circle(1.0);
        \draw[thick] (0,0) circle(0.9);
        
        % draw cusps
        \foreach \x in {0,30,...,330} {
                % lines from center to point
                \draw[gray] (0,0) -- (\x:1);
                % dots at each point
                %\filldraw[black] (\x:1cm) circle(0.4pt);
                % draw each angle in degrees
               %\draw (\x:0.6cm) node[fill=white] {$\x^\circ$};
        }
        
        % draw center of circle
        \fill[white] (0,0) circle(0.1);
        \draw[thick] (0,0) circle(0.1);
        
        % draw sign labels
        \foreach \sign/\label in {
        15 /\Aries, 45 /\Taurus, 75 /\Gemini,
        105 /\Cancer, 135 /\Leo, 165 /\Virgo,
        195 /\Libra, 225 /\Scorpio, 255 /\Sagittarius,
        285 /\Capricorn, 315 /\Aquarius, 345 /\Pisces}
        \draw (\sign:0.95) node {\label};
        
        % draw Asc and MC lines
        \draw [->, red, ultra thick] 
        	(185:1.2)--(5:1.2cm)node [above] {\ASC $ 5^\circ$};
        
        \draw [->, blue, ultra thick] 
        	(100:1.2)--(280:1.2)node [left] {\MC $ 10^\circ$};
        	
        % colour profitable places
        % 	arc(start angle : delta angle : radius)
        \scoped[on background layer]
        	{\filldraw[green!20!white]
        		(0,0) -- (5:.9) arc (5:32:.9) -- cycle;
		  	  \filldraw[green!05!white]
		  		(0,0) -- (32:.9) arc (32:64:.9) -- cycle;}
        	       	        
        % draw planet in houses at Exaltation degrees
        \node[rotate=19] at (19:0.75) {\Sun $19^\circ$};        
        \node[rotate=33] at (33:0.75) {\Moon $3^\circ$};
        \node[rotate=165] at (165:0.75) {\rotatebox{180}
        	{\Mercury $15^\circ$}};     
	     \node[rotate=357] at (357:0.75){\Venus $27^\circ$};
	     \node[rotate=298] at (298:0.75) {\Mars $28^\circ$};
        \node[rotate=105] at (105:0.75) {\Jupiter $15^\circ$};
        \node[rotate=201] at (201:0.75) 
        	{\rotatebox{180}{\Saturn $21^\circ$}};
     		  
\end{tikzpicture}
\end{document}
