\begin{tikzpicture}[scale=3.5]
\begin{scope}[rotate=165]
	% draw numbered, fixed house positions
	\foreach \sign/\label in {
		0 / 1, 
		30/ 2, 
		60 / 3,
		90 / 4, 
		120 / 5, 
		150 / 6,
		180/ 7 ,

\begin{scope}[rotate=-15]
	% --- Draw the basic chart
	% draw the circles
	\draw[thick] (0,0) circle(1.05);
	\draw[thick] (0,0) circle(0.85);
	
	% draw cusps
	\foreach \x in {0,30,...,330} {
	        % lines from center to point
	        \draw[gray] (0,0) -- (\x:1.05);
	}
	
	% draw center of circle
	\fill[white] (0,0) circle(0.15);
	\draw (0,0) circle(0.15);
	\draw (0,0) circle(0.25);
	
	% draw sign labels
	\foreach \sign/\label in {
	15 /\Aries, 45 /\Taurus, 75 /\Gemini,
	105 /\Cancer, 135 /\Leo, 165 /\Virgo,
	195 /\Libra, 225 /\Scorpio, 255 /\Sagittarius,
	285 /\Capricorn, 315 /\Aquarius, 345 /\Pisces}
	\draw (\sign:0.95) node {\label};
	
	% --- draw Asc and MC lines
	\draw [->, red] 
	    (15:1.2)--(195:1.2cm)
	    node [above] {\ASC}; % $ 0^\circ$};
	
	%\draw [->, blue, ultra thick] 
	%    (100:1.2)--(280:1.2)node [left] {\MC $ 10^\circ$};	
	
	% -- Colour house
	 % -- 	arc(start angle : delta angle : radius)
     \scoped[on background layer]\filldraw[green!10]
     	(0,0) -- (240:.85) arc (240:270:.85);
 	 \scoped[on background layer]\filldraw[yellow!20]
     	(0,0) -- (120:.85) arc (120:150:.85);
	
	% --- draw planets
	\node         at (214:0.75)  {\Sun};      
	\node [below] at (216:0.75) {\Mercury};     
	\node [below, blue] at (225:0.70) {\Large\Mars};    
	\node [blue]   at (106:0.75) {\LARGE\Moon};     
	\node [blue]   at (190:0.75) {\large\Venus};    
	\node [right] at (243:0.75) {\Jupiter}; 
	\node         at (310:0.75) {\Saturn};   
	\node         at (315:0.5)  {\Fortune};
	
	% -- draw aspect
	\draw [black!50!green] (106:0.65)-- node[above, left]
		{\Trine} (230:0.65);
		
\end{scope}
\end{tikzpicture}
