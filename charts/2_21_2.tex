\begin{tikzpicture}[rotate=90, scale=3.5]
	% --- Draw the basic chart
	% draw the circles
	\draw[thick] (0,0) circle(1.0);
	\draw[thick] (0,0) circle(0.9);
	
	% draw cusps
	\foreach \x in {0,30,...,330} {
	        % lines from center to point
	        \draw[gray] (0,0) -- (\x:1);
	}
	
	% draw center of circle
	\fill[white] (0,0) circle(0.1);
	\draw[thick] (0,0) circle(0.1);
	
	% draw sign labels
	\foreach \sign/\label in {
	15 /\Aries, 45 /\Taurus, 75 /\Gemini,
	105 /\Cancer, 135 /\Leo, 165 /\Virgo,
	195 /\Libra, 225 /\Scorpio, 255 /\Sagittarius,
	285 /\Capricorn, 315 /\Aquarius, 345 /\Pisces}
	\draw (\sign:0.95) node {\label};
	
	% --- draw Asc and MC lines
	\draw [->, red, ultra thick] 
	    (285:1.2)--(105:1.2cm)
	    node [above] {\ASC}; % $ 0^\circ$};
	
	\draw [->, blue, ultra thick] 
	    (195:1.2)--(15:1.2)node [left] {\MC}; %$ 10^\circ$};	
	
	% --- draw planets
	\node         at (35:0.8)   {\Sun};      %$ 1^\circ$}; 
	\node [below] at (40:0.8)   {\Mercury};  %$13^\circ$};     
	\node         at (10:0.75)  {\Moon};     %$16^\circ$};
	\node         at (280:0.75) {\Jupiter}; %$ 3^\circ$};
	\node         at (95:0.75)  {\Saturn};   %$10^\circ$};
	\node [below] at (100:0.75) {\Mars};     %$27^\circ$}; 
	\node [below] at (110:.75)  {\Venus};    %$21^\circ$};	    	
	\node         at (70:0.75)  {\Fortune};
	
	% -- Placements given in Greek Horoscopes
	\node	[{red}] at (135:0.8)  {[\Saturn]};
	\node	[{red}] at (165:0.8)  {[\Jupiter]};
	
	% -- draw aspects
	
	
\end{tikzpicture}
